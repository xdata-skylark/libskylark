Coding style¶

Please make sure to use one of the provided editor configurations when contributing to libskylark. The following examples can be found in the doc/script folder:

" .vimrc
" Author: Yves Ineichen
" Very basic vim configuration file corresponding to skylark code style guide.
" ---------------------------------------------------------------------------
" basic options

set autoindent smartindent    " auto/smart indent
set expandtab                 " expand tabs to spaces
set smarttab                  " tab and backspace are smart
set tabstop=4                 " 4 spaces
set shiftwidth=4              " 4 spaces
set backspace=indent,eol,start
set tw=78                     " default textwidth is a max of 78
set list                      " enable custom list chars
set listchars=tab:▸\ ,eol:¬   " explicitly show tabs and eol
set colorcolumn=+1            " show line where text should wrap
set encoding=utf-8
set hidden
set magic


" ---------------------------------------------------------------------------
" auto load extensions for different file types
if has('autocmd')
        filetype plugin indent on
        syntax on

        " strip all whitespaces at save
        autocmd BufWritePre * :%s/\s\+$//e

endif
" ---------------------------------------------------------------------------

;; .emacs
;; Author: Haim Avron
;; Very basic emacs configuration file to help working with skylark 
;; code style.

; The following will highlight beyond column 80 in C and C++, so you will know you are
; bad boy if you go over.
(require 'whitespace)
(setq whitespace-style '(face empty tabs lines-tail trailing))
(add-hook 'c++-mode-hook (lambda() (interactive) (whitespace-mode)))
(add-hook 'c-mode-hook (lambda() (interactive) (whitespace-mode)))

; Indentation setup
(defun skylark-indent-setup ()
  (setq c-basic-offset 4)
  (c-set-offset 'arglist-cont-nonempty '+)
  (c-set-offset 'arglist-cont '+)
  (c-set-offset 'inextern-lang 0)
  (c-set-offset 'innamespace 0))

(add-hook 'c-mode-hook 'skylark-indent-setup)
(add-hook 'c++-mode-hook 'skylark-indent-setup)

(setq-default indent-tabs-mode nil) ; always replace tabs with spaces

;;(setq-default c-basic-offset 4) ; and especially for c mode
